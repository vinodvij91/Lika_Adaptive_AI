# /api/structures/openfold3
@app.post("/api/structures/openfold3")
async def predict_structure(request: dict):
    """
    General OpenFold3 structure prediction
    Supports: protein, DNA, RNA, ligand (SMILES)
    """
    payload = {
        "input_id": request["input_id"],
        "molecules": request["molecules"]
        # molecules: [
        #   {"type": "protein", "sequence": "MTEYK..."},
        #   {"type": "ligand", "smiles": "CC1=CC=C..."}
        # ]
    }
    
    resp = requests.post(OPENFOLD3_URL, json=payload)
    structure = resp.json()
    
    # Save to storage
    structure_id = save_structure(structure["cif"])
    
    return {
        "structure_id": structure_id,
        "confidence": structure["confidence"],
        "pdb_url": f"/structures/{structure_id}.pdb",
        "cif_url": f"/structures/{structure_id}.cif"
    }

# /api/materials/predict_properties
@app.post("/api/materials/predict_properties")
async def predict_properties(request: dict):
    """
    BioNeMo material property prediction
    Supports: polymers, biomaterials, composites
    """
    materials = [get_material(id) for id in request["material_ids"]]
    properties_to_predict = request["properties"]
    
    predictions = []
    
    for material in materials:
        # Prepare input for BioNeMo
        bionemo_input = {
            "material_id": material.id,
            "smiles": material.smiles,
            "sequence": material.sequence,  # For biomaterials
            "composition": material.composition,
            "properties": properties_to_predict,
            "context": request.get("context", "general")
        }
        
        # Call BioNeMo
        result = requests.post(
            BIONEMO_MATERIALS_URL,
            json=bionemo_input
        ).json()
        
        predictions.append({
            "material_id": material.id,
            "predictions": result["predictions"],
            "confidence": result["confidence"]
        })
    
    return {"predictions": predictions}

    })

@app.post("/api/materials/spider_silk/generate_variants")
async def generate_silk_variants(base_id: str, num_variants: int):
    """
    Generate MaSp1/2 sequence variants
    """
    base_material = get_material(base_id)
    base_seq = base_material.sequence
    
    variants = []
    
    for i in range(num_variants):
        # Variant strategies:
        # 1. Repeat block expansion (GPGXX motifs)
        # 2. Poly-A stretch modification
        # 3. Conservative mutations in spacer regions
        
        variant_seq = apply_variant_strategy(base_seq, i)
        
        variant = {
            "id": f"{base_id}_var_{i:03d}",
            "name": f"{base_material.name} Variant {i+1}",
            "category": "biomaterial",
            "sequence": variant_seq,
            "parent_id": base_id,
            "mutation_strategy": "repeat_expansion",
            "tags": ["spider_silk", "variant", "computational"]
        }
        
        variants.append(create_material(variant))
    
    return {"variants": variants}

@app.post("/api/materials/pfas/screen")
async def screen_pfas_replacements(request: dict):
    """
    Screen PFAS-free polymer library
    """
    # Load candidate library
    candidates = load_material_library(request["library_id"])
    
    # Filter stage 1: PFAS exclusion
    pfas_free = [m for m in candidates if is_pfas_free(m)]
    
    # Filter stage 2: Property constraints
    use_case = request["use_case"]  # "seals", "gaskets", "coatings"
    constraints = get_constraints(use_case)
    
    viable = [m for m in pfas_free if meets_constraints(m, constraints)]
    
    # Predict properties via BioNeMo
    predictions = await predict_properties({
        "material_ids": [m.id for m in viable],
        "properties": [
            "thermal_stability_C",
            "chemical_resistance_index",
            "friction_coefficient",
            "dielectric_constant"
        ]
    })
    
    # Rank by multi-objective
    ranked = rank_materials(predictions, constraints)
    
    return {
        "initial_library_size": len(candidates),
        "pfas_free_count": len(pfas_free),
        "viable_count": len(viable),
        "top_candidates": ranked[:20]
    }
